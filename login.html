document.getElementById('loginForm').addEventListener('submit', async function(event) { event.preventDefault(); const username = document.getElementById('username').value.trim().toLowerCase(); const password = document.getElementById('password').value; const response = await fetch('/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ username, password }), }); if (response.ok) { alert('Login successful!'); window.location.href = 'dashboard.html'; } else { alert('Login failed'); } });
